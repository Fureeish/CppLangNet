#include <algorithm>
#include <iostream>
#include <vector>

int main()
{
    const std::vector<int> onlyEvens = {
        2, 4, 6, 8, 10
    };

    auto positionOf1 = std::ranges::find(onlyEvens, 1);

    if (positionOf1 != onlyEvens.end())
    {
        std::cout << "odd element found: " << *positionOf1;
    }
    else
    {
        std::cout << "no odd elements found";
    }
}
