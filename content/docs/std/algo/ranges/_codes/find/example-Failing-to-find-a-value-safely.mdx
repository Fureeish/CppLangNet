#include <algorithm>
#include <iostream>
#include <vector>

auto main() -> int
{
    auto const onlyEvens = std::vector<int>{
        2, 4, 6, 8, 10
    };

    auto const positionOf1 = std::ranges::find(onlyEvens, 1);

    if (positionOf1 != onlyEvens.end())
    {
        std::cout << "odd element found: " << *positionOf1;
    }
    else
    {
        std::cout << "no odd elements found";
    }
}
