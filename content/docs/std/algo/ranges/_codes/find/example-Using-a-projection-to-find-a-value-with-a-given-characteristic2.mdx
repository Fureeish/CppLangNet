#include <algorithm>
#include <iostream>
#include <vector>

int main()
{
    const std::vector<int> numbers = {
        5, 6, 7, 8
    };

    const auto byModulus5 = [](const int n) {
        return n % 5;
    };

    const auto withRemainder2 = std::ranges::find(numbers, 2, byModulus5);
    std::cout << *withRemainder2;
}
